CREATE TABLE camps(
    camp_id INT(11) PRIMARY KEY
)

CREATE TABLE invoices(
    invoice_id INT(11) PRIMARY KEY,
    guest_id INT(11) NOT NULL,
    paymentMethod VARCHAR(255) NOT NULL,
    reserve_id INT(11) NOT NULL
)

CREATE TABLE reserves(
    reserve_id INT(11) PRIMARY KEY,
    guest_id INT(11) NOT NULL,
    camp_id INT(11) NOT NULL,
    camping_style TINYINT(1) NOT NULL,
    electricity TINYINT(1) NOT NULL,
    price INT(11) NOT NULL,
    reserve_start DATE NOT NULL,
    reserve_end DATE NOT NULL
)

CREATE TABLE guests(
    guest_id INT(11) PRIMARY KEY,
    guest_name VARCHAR(255) NOT NULL,
    guest_email VARCHAR(255) NOT NULL,
    guest_phone INT(11) NOT NULL
)
